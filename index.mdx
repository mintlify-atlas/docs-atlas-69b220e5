      ```bash pnpm
      pnpm add valleys
      ```
      </CodeGroup>
    </Step>

    <Step title="Import validators">
      Import the validators and validation function you need:
      
      ```typescript
      import { validate, object, string, number, boolean, array } from 'valleys';
      ```
    </Step>

    <Step title="Validate your data">
      Create validators and use them to validate runtime data:
      
      ```typescript
      const userValidator = object({
        id: number(),
        name: string(),
        email: string(),
        isActive: boolean(),
        tags: array(string())
      });

      const data: unknown = getUserData(); // untrusted data from API
      
      validate(data, userValidator);
      
      // TypeScript now knows data is:
      // { id: number; name: string; email: string; isActive: boolean; tags: string[] }
      console.log(data.name.toUpperCase()); // ✅ No type errors
      ```
    </Step>
  </Steps>
